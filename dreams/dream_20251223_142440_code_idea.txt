def calculate_entropy(data_stream):
    from collections import Counter
    import math
    
    # Calculate the frequency of each data point
    counts = Counter(data_stream)
    
    # Calculate the probabilities
    probabilities = [count / len(data_stream) for count in counts.values()]
    
    # Calculate Shannon entropy
    entropy = -sum(p * math.log(p, 2) for p in probabilities if p > 0)
    
    return entropy
