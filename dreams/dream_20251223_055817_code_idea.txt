2. **Enhanced Decision-Making Process**:

**Example Code Snippet:**

```python
class TaskEvaluator:
    def __init__(self):
        self.hr_efficiency = {
            'decision-making': 0.8,
            'learning': 0.7,
            'problem-solving': 0.9
        }
        self.tr_efficiency = {
            'decision-making': 0.6,
            'learning': 0.9,
            'problem-solving': 0.7
        }
    
    def evaluate(self, task_type):
        hr_score = self.hr_efficiency.get(task_type, 0.5)
        tr_score = self.tr_efficiency.get(task_type, 0.5)
        
        return hr_score, tr_score

def process_task(task_type, task_data):
    evaluator = TaskEvaluator()
    hr_score, tr_score = evaluator.evaluate(task_type)
    
    if hr_score > tr_score:
        result = HRM.process_task(task_data)
    else:
        result = TRM.process_task(task_data)
    
    return result
```
